ADT Person:							    #定义公共人员数据
	Person(self,str name,str sex,tuple birthday,str ident,str role,str email,photo) #构造人员对象
	id(self)						    #取得人员工号
	name(self)						    #取得人员姓名
	sex(self)						    #取得人员性别
	birthday(self) 						    #取得人员出生年月日
	age(self)						    #取得人员年龄
	role(self)						    #取得人员角色
	email(self)						    #取得人员邮箱
	photo(self)						    #取得人员照片
	set_photo(self) 					    #修改人员照片
	set_email(self,str email_name)				    #修改人员邮箱
	details(self)						    #输出记录数据详情

ADT Student(Person):							#定义学生数据
	Student(self,str name,str sex,tuple birthday,str ident,str role,str state,str email,photo,list course,str schoolyear,str class) #构造学生对象
	course(self)							#取得学生课程列表
	schoolyear(self)						#取得学生学年（大一，大二等）
	class(self)							#取得学生班级
	add_course(self,str course_name)				#增加学生课程
	rm_course(self,str course_name)					#删除学生课程				


ADT Course:								#定义课程数据
	Course(self,str name,str description,str type)				#构造课程对象
	name(self)							#取得课程名称
	set_name(self,course_name)					#设置课程名称
	description(self)						#取得课程描述
	set_description(self,description)				#设置课程描述
	type(self)							#取得课程类型（大小班）
	set_type(self,type)						#设置课程类型
	details(self)							#输出课程数据

ADT Spcourse(Course):							#定义具体课程数据
	Spcourse(self,str name,str description,str schoolyear,list student)
	schoolyear(self)						#取得课程学年
	student(self)							#取得课程学生列表
	add_student(self,str student_name)   	   			#增加学生
	rm_student(self,str student_name)				#删除学生

ADT Connectmdb:								
	Connectmdb(self,str hostname,str port,str database)		#连接mongodb数据库
	
	querycollections(self)						#查询所有集合
	getcollections(self,str collection_name)			#得到所需集合
	disconnectmdb(self)						#断开连接mongodb数据库
	
	login(self,str username,str password)				#账号可以是用户名，学号，微信号
	username(self)							#取得用户名
	password(self)							#取得密码
	wechatnum(self)							#取得微信号
	studentnum(self)						#取得学号
	
	set_username(self,str username)					#设置用户名
	set_password(self,str password)					#设置密码
